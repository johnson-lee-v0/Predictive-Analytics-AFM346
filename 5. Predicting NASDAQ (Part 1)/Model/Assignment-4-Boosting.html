Base Recipe

boost_base_recipe<-
  recipe(Up_Down~ ., data = data_train)%>%
  step_zv(all_numeric(), -all_outcomes()) %>%
  step_corr(all_numeric(), -all_outcomes())
boost_enhanced_recipe <-

boost_model <-
    boost_tree(trees = tune(),

    set_engine('xgboost')
