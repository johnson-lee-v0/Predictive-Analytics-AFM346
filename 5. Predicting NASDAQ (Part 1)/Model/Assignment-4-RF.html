data_folds <- data_train %>%

Base Recipe

rf_base_recipe<-
  recipe(Up_Down~ ., data = data_train)%>%
  step_zv(all_numeric(), -all_outcomes()) %>%
  step_corr(all_numeric(), -all_outcomes())
rf_enhanced_recipe <- 
 rf_base_recipe%>%
  step_bs(-Up_Down)

#Create Model

rf_model <-
    rand_forest(trees = tune(),
                mtry = tune()) %>%
    set_mode(('classification')) %>%
    set_engine('ranger', importance = "impurity")
